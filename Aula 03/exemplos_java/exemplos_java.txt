public static int somarPares(int limite) {
    int soma = 0;
    int numero = 2;
    while (numero <= limite) {
        soma += numero;
        numero += 2;
    }
    return soma;
}

// Exemplo 3: Padr√£o de asteriscos usando FOR aninhado
public static void exibirPadrao(int linhas) {
    for (int i = 1; i <= linhas; i++) {
        for (int j = 1; j <= i; j++) {
            System.out.print("* ");
        }
        System.out.println();
    }
}
